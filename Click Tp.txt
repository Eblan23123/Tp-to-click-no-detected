local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")
local TweenService = game:GetService("TweenService")

-- === НАСТРОЙКИ ===
local MY_LINK = "https://link-target.net/4164566/kBdEjDTjl7fJ"
local CORRECT_KEY = "FLORIK_2026" -- Установи здесь ключ, который игроки увидят в конце твоей ссылки
local PASTEBIN_URL = "https://pastebin.com/raw/gDML9pyb" -- Твой основной чит

-- === СОЗДАНИЕ ОКНА РЕГИСТРАЦИИ (UI) ===
local screenGui = Instance.new("ScreenGui", Players.LocalPlayer:WaitForChild("PlayerGui"))
screenGui.Name = "FlorikRegistration"

local main = Instance.new("Frame", screenGui)
main.Size = UDim2.new(0, 350, 0, 200)
main.Position = UDim2.new(0.5, -175, 0.5, -100)
main.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
main.BorderSizePixel = 0
main.ClipsDescendants = true

local corner = Instance.new("UICorner", main)
corner.CornerRadius = UDim.new(0, 10)

-- Заголовок
local title = Instance.new("TextLabel", main)
title.Size = UDim2.new(1, 0, 0, 50)
title.Text = "FLORIK HUB | REGISTRATION"
title.TextColor3 = Color3.new(1, 1, 1)
title.BackgroundTransparency = 1
title.Font = Enum.Font.GothamBold
title.TextSize = 18

-- Поле ввода
local inputField = Instance.new("TextBox", main)
inputField.Size = UDim2.new(0.8, 0, 0, 40)
inputField.Position = UDim2.new(0.1, 0, 0.35, 0)
inputField.BackgroundColor3 = Color3.fromRGB(40, 40, 40)
inputField.Text = ""
inputField.PlaceholderText = "Вставь ключ здесь..."
inputField.TextColor3 = Color3.new(1, 1, 1)
local inputCorner = Instance.new("UICorner", inputField)

-- Кнопка ПРОВЕРИТЬ
local verifyBtn = Instance.new("TextButton", main)
verifyBtn.Size = UDim2.new(0.35, 0, 0, 40)
verifyBtn.Position = UDim2.new(0.1, 0, 0.65, 0)
verifyBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 70)
verifyBtn.Text = "ПРОВЕРИТЬ"
verifyBtn.TextColor3 = Color3.new(1, 1, 1)
local vCorner = Instance.new("UICorner", verifyBtn)

-- Кнопка ПОЛУЧИТЬ КЛЮЧ
local getBtn = Instance.new("TextButton", main)
getBtn.Size = UDim2.new(0.45, 0, 0, 40)
getBtn.Position = UDim2.new(0.5, 0, 0.65, 0)
getBtn.BackgroundColor3 = Color3.fromRGB(150, 80, 0)
getBtn.Text = "ПОЛУЧИТЬ КЛЮЧ"
getBtn.TextColor3 = Color3.new(1, 1, 1)
local gCorner = Instance.new("UICorner", getBtn)

-- === ЛОГИКА РАБОТЫ ===

-- Копирование ссылки
getBtn.MouseButton1Click:Connect(function()
    if setclipboard then
        setclipboard(MY_LINK)
        getBtn.Text = "СКОПИРОВАНО!"
        task.wait(2)
        getBtn.Text = "ПОЛУЧИТЬ КЛЮЧ"
    else
        getBtn.Text = "ОШИБКА КОПИРОВАНИЯ"
    end
end)

-- Проверка ключа
verifyBtn.MouseButton1Click:Connect(function()
    if inputField.Text == CORRECT_KEY then
        verifyBtn.Text = "УСПЕХ!"
        verifyBtn.BackgroundColor3 = Color3.fromRGB(0, 200, 0)
        
        -- Анимация закрытия
        main:TweenPosition(UDim2.new(0.5, -175, 1.2, 0), "In", "Back", 0.5)
        task.wait(0.6)
        screenGui:Destroy()

        -- ЗАГРУЗКА ТВОЕГО ОСНОВНОГО ПАСТБИНА (Полет + Меню)
        local success, err = pcall(function()
            loadstring(game:HttpGet(PASTEBIN_URL))()
        end)
        
        if not success then
            warn("Ошибка загрузки пастбина: " .. tostring(err))
        end
    else
        inputField.Text = ""
        inputField.PlaceholderText = "НЕВЕРНЫЙ КЛЮЧ!"
        verifyBtn.BackgroundColor3 = Color3.fromRGB(150, 0, 0)
        task.wait(1)
        verifyBtn.BackgroundColor3 = Color3.fromRGB(0, 150, 70)
        verifyBtn.Text = "ПРОВЕРИТЬ"
    end
end)
